Milestone 1:
CREATE TABLE SALESMAN (SNO INTEGER PRIMARY KEY, S_NAME CHAR (30), START_YEAR DATE, DEPTNO VARCHAR (10));
CREATE TABLE TRIP (TNO INTEGER PRIMARY KEY, FROM_CITY CHAR (20), TO_CITY CHAR (20), DEPARTURE_DATE DATE, RETURN DATE);
CREATE TABLE DEPT (DEPTNO VARCHAR(10) PRIMARY KEY PRIMARY KEY, DEPT_NAME CHAR(20));
CREATE TABLE EXPENSE (EID INTEGER PRIMARY KEY, AMOUNT MONEY);
ALTER TABLE SALESMAN ADD FOREIGN KEY(deptno) REFERENCES DEPT(deptno);
ALTER TABLE TRIP ADD COLUMN SNO INT REFERENCES SALESMAN(SNO);
ALTER TABLE TRIP ADD COLUMN EID INT REFERENCES EXPENSE(EID);
